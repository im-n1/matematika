\section{Obecné}

\begin{description}
    \item{\boldmath{$\neg x$}}\\
        negace - opak výrazu
    \item{\boldmath{$x \land y$}}\\
        konjunkce - a, současně
    \item{\boldmath{$x \lor y$}}\\
        disjunkce - nebo
    \item{\boldmath{$x \Rightarrow y$}}\\
        implikace - z $x$ plyne $y$
    \item{\boldmath{$x \Leftrightarrow y$}}\\
        ekvivalence - $x$ je ekvivalentní s $y$
\end{description}
